#### Algorithm

**[题目]** ：https://leetcode-cn.com/problems/remove-duplicate-node-lcci/

**[题解]** ：

  ```c++
class Solution {
public:
    ListNode *removeDuplicateNodes(ListNode *head) {
        if (head == nullptr) {
            return nullptr;
        }
        unordered_map<int, int> v;
        v[head->val]++;
        ListNode *cur;
        cur = head;
        while (cur->next) {
            if (v[cur->next->val]) {
                cur->next = cur->next->next;
            } else {
                v[cur->next->val]++;
                cur = cur->next;
            }
        }
        return head;
    }
};
  ```
